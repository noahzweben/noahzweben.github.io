 40   public void reverseLinkedListIteratively() {
 41     Node prevNode = null;
 42     Node nextNode;
 43     Node currentNode = head;
 44 
 45     while (currentNode != null) {
 46       nextNode = currentNode.next;
 47       currentNode.next = prevNode;
 48 
 49       prevNode = currentNode;
 50       currentNode = nextNode;
 51     }
 52     head = prevNode;
 53   }

 55   public void reverseLinkedListRecursively(Node node) {
 56     if (node == null) return;
 57     if (node.next==null) {
 58       head = node;
 59       return;
 60     }
 61 
 62     reverseLinkedListRecursively(node.next);
 63     node.next.next = node;
 64     node.next = null;
 65 
 66   }

 69   public boolean detectLoops(){
 70     Node slow = head;
 71     Node fast = head.next.next;
 72 
 73     while (slow != null && fast != null) {
 74       if (slow == fast) return true;
 75       slow = slow.next;
 76       fast = fast.next.next;
 77     }
 78     return false;

 85     Node current = head;
 86     Map<Integer,Integer> map = new HashMap<>();
 87     
 88     while (current != null) {
 89       if (map.containsKey(current.value)) {
 90         remove(current);
 91         current=current.next;
 92       }
 93       else {
 94         map.put(current.value,null);

135       // Switches the previous and next pointers

137       currentNode.next = currentNode.prev;
