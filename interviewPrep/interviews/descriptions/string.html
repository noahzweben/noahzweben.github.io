 53     // Base Cases
 54     if (m == 0) return true;
 55     if (n == 0) return false;
 56  
 57     // If last characters of two strings are matching
 58     if (str1[m-1] == str2[n-1])
 59         return isSubSequence(str1, str2, m-1, n-1);
 60  
 61     // If last characters are not matching
 62     return isSubSequence(str1, str2, m, n-1);

 95 ///////////////////////////////////////////////// push zeroes to the right
 96 void pushZerosToEnd(int arr[], int n)
 97 {
 98     int count = 0;  // Count of non-zero elements
 99  
100     // Traverse the array. If element encountered is non-
101     // zero, then replace the element at index 'count' 
102     // with this element
103     for (int i = 0; i < n; i++)
104         if (arr[i] != 0)
105             arr[count++] = arr[i]; // here count is 
106                                    // incremented
107  
108     // Now all non-zero elements have been shifted to 
109     // front and  'count' is set as index of first 0. 
110     // Make all elements 0 from count to end.
111     while (count < n)
112         arr[count++] = 0;

117 //////////////////////////////////// Longest Palindrome
118  public String longestPalindrome(String s) {
119   if (s.isEmpty()) {
120     return null;
121   }
122  
123   if (s.length() == 1) {
124     return s;
125   }
126  
127   String longest = s.substring(0, 1);
128   for (int i = 0; i < s.length(); i++) {
129     // get longest palindrome with center of i
130     String tmp = helper(s, i, i);
131     if (tmp.length() > longest.length()) {
132       longest = tmp;
133     }
134  
135     // get longest palindrome with center of i, i+1
136     tmp = helper(s, i, i + 1);
137     if (tmp.length() > longest.length()) {
138       longest = tmp;
139     }
140   }
141   return longest;
142 }
143 // Given a center, either one letter or two letter, 
144 // Find longest palindrome
145 public String helper(String s, int begin, int end) {
146   while (begin >= 0 && end <= s.length() - 1 && s.charAt(begin) == s.charAt(end)) {
147     begin--;
148     end++;
149   }
150   return s.substring(begin + 1, end);
151 }
