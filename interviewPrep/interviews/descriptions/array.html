 18 ////////////////////////////////////////////////// Check if an array has a SubArray which sums to 0
 19         // Returns true if arr[] has a subarray with sero sum
 20     static Boolean printZeroSumSubarray(int arr[])
 21     {
 22         // Creates an empty hashMap hM
 23         HashMap<Integer, Integer> hM = new HashMap<Integer, Integer>();
 24          
 25         // Initialize sum of elements
 26         int sum = 0;        
 27          
 28         // Traverse through the given array
 29         for (int i = 0; i < arr.length; i++)
 30         {   
 31             // Add current element to sum
 32             sum += arr[i];
 33              
 34             // Return true in following cases
 35             // a) Current element is 0
 36             // b) sum of elements from 0 to i is 0
 37             // c) sum is already present in hash map
 38             if (arr[i] == 0 || sum == 0 || hM.get(sum) != null)                            
 39                return true;
 40              
 41             // Add sum to hash map
 42             hM.put(sum, i);
 43         }    
 44          
 45         // We reach here only when there is no subarray with 0 sum
 46         return false;
 47     }

 53         //////////////////////////////////////////////////  Subarray with sum==sum
 54 /* Returns true if the there is a subarray of arr[] with sum equal to 'sum'
 55    otherwise returns false.  Also, prints the result */
 56 int subArraySum(int arr[], int n, int sum)
 57 {
 58     /* Initialize curr_sum as value of first element
 59        and starting point as 0 */
 60     int curr_sum = arr[0], start = 0, i;
 61  
 62     /* Add elements one by one to curr_sum and if the curr_sum exceeds the
 63        sum, then remove starting element */
 64     for (i = 1; i <= n; i++)
 65     {
 66         // If curr_sum exceeds the sum, then remove the starting elements
 67         while (curr_sum > sum && start < i-1)
 68         {
 69             curr_sum = curr_sum - arr[start];
 70             start++;
 71         }
 72  
 73         // If curr_sum becomes equal to sum, then return true
 74         if (curr_sum == sum)
 75         {
 76             printf ("Sum found between indexes %d and %d", start, i-1);
 77             return 1;
 78         }
 79  
 80         // Add this element to curr_sum
 81         if (i < n)
 82           curr_sum = curr_sum + arr[i];
 83     }
 84  
 85     // If we reach here, then no subarray
 86     printf("No subarray found");
 87     return 0;
 88 }

 91 /////////////////////////// Max sum of subArray
 92   public int maxSubArray(int[] A) {
 93        int newsum=A[0];
 94        int max=A[0];
 95        for(int i=1;i<A.length;i++){
 96            newsum=Math.max(newsum+A[i],A[i]);
 97            max= Math.max(max, newsum);
 98        }
 99        return max;
100     }
