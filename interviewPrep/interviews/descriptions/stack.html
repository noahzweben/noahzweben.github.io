1 //sort a stack w/ another stack
 2 public static Stack<Integer> sort(Stack<Integer> s) {
 3     Stack<Integer> r = new Stack<Integer>();
 4 
 5     while(!s.isEmpty()) {
 6         int tmp = s.pop();
 7         while(!r.isEmpty() && r.peek() > tmp) {
 8             s.push(r.pop());
 9         }
10         r.push(tmp);
11     }
12     return r;
13 }

15 // sort a stack w/ recursion
16 sortStack(stack S)
17     if stack is not empty:
18         temp = pop(S);  
19         sortStack(S); 
20         sortedInsert(S, temp);
21 
22 sortedInsert(Stack S, element)
23     if stack is empty OR element > top element
24         push(S, elem)
25     else
26         temp = pop(S)
27         sortedInsert(S, element)
28         push(S, temp)

32 Push(int x) // inserts an element x to Special Stack 
33 1) push x to the first stack (the stack with actual elements)
34 2) compare x with the top element of the second stack (the auxiliary stack). Let the top element be y.
35 …..a) If x is smaller than y then push x to the auxiliary stack.
36 …..b) If x is greater than y then push y to the auxiliary stack.
37 
38 int Pop() // removes an element from Special Stack and return the removed element 
39 1) pop the top element from the auxiliary stack.
40 2) pop the top element from the actual stack and return it.
41 
42 The step 1 is necessary to make sure that the auxiliary stack is also updated for future operations.
